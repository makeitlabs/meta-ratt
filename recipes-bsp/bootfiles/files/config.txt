# force 1024x768

# set gpu memory

# also contains gpio-keys mapping for keypad keys

dtoverlay=ratt-ioexpander

# i2s audio
dtparam=i2s=on
dtoverlay=hifiberry-dac
dtoverlay=i2s-mmap

# enable uart for serial port
enable_uart=1
